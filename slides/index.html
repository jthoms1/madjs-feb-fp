What is Functional Programming (FP)?

In computer science, functional programming is a programming paradigm—a style of building the structure and elements of computer programs—that treats computation as the evaluation of mathematical functions and avoids changing-state and mutable data.
- wikipedia

What is Functional Programming (FP)?

Functional programming is a style of building programs where application data flows through functions.
- me

Why use FP?

  • Concurrency
  • Easier to reason about
  • Code is easy to reuse
  • Testing gets a lot easier

var people = [
  { name: 'Josh', age: 35, occupation: 'Developer' },
  { name: 'Lindsey', age: 35, occupation: 'Project Manager' },
  { name: 'Ellie', age: 2, occupation: 'Baby'},
  { name: 'Addie', age: 1, occupation: 'Baby'}
];
var names = [];

for (var i = 0; i < people.length; i += 1) {
  names.push(people[i].name);
}
console.log(names); // ['Josh', 'Lindsey', 'Ellie', 'Addie']

var people = [...];
function getAttribute(attributeName) {
  return function(obj) {
    return obj[attributeName];
  }
}
function strLength(str) {
  return str.length;
}
function evenOrOdd(num) {
  return (num % 2) === 1 ? 'odd' : 'even';
}
function compose() {
  var fns = arguments;

  return function (result) {
    for (var i = fns.length - 1; i > -1; i -= 1) {
      result = fns[i].call(this, result);
    }
    return result;
  };
};
var isNameEvenOrOddLength = compose(evenOrOdd, strLength, getAttribute('name'))
console.log(people.map(isNameEvenOrOddLength));
// ['even', 'odd', 'odd', 'odd']
  • Function Composition is combining multiple functions where the output of one function is the input for another.
